Actor BDW_EmptyClip : BDW_CasingBase {
  Height 12
  Radius 9
  Speed 4
  Scale 1.01
  +DOOMBOUNCE
  -NOGRAVITY
  +WINDTHRUST
  +CLIENTSIDEONLY
  +MOVEWITHSECTOR
  +MISSILE
  -DROPOFF
  +NOTELEPORT
  +FORCEXYBILLBOARD
  +NOTDMATCH
  +GHOST
  -NoLiftDrop
  +NOBLOCKMAP
  Mass 1
  //k8: not yet:SeeSound "weapons/largemagdrop"
  SeeSound "weapons/largemagdrop1"
  States {
    Spawn:
      TNT1 A 0
      ECLI ABCDEFGHABCDEFGHABCDEFGHABCDEFGHABCDEFGH 4
        {
          if (!A_k8JumpIfOnFloorSky("Disappear")) {
            A_k8JumpIfOnLiquidFloor("Death");
          }
        }
      Stop

    Disappear:
      TNT1 A 0
      Stop

    Death:
      TNT1 A 0 A_ChangeFlag("DOOMBOUNCE", false)
      ECLI G 180
      "####" "#" 0 A_ChangeFlag("NoInteraction", true)
      "####" "#" 0
        {
          if (!A_k8JumpIfOnFloorSky("Disappear")) {
            A_k8JumpIfOnLiquidFloor("Stopping");
          }
        }
      //"####" "#" random(800, 1200)
      "####" "#" 0 A_SetTics(GetCVar("K8BDWOpt_EmptyClipTime")*35*60+10)
      //"####" "#" 0 //A_JumpIfInventory("LowGraphicsMode", 1, "Stopping")
    FadeOut:
      "####" "#" random(10, 15) A_FadeOut(0.01)
      Loop

   Stopping:
      "####" "#" 0 A_ChangeFlag("NoInteraction", true)
      "####" "#" 0 A_ChangeVelocity(0, 0, -14, CVF_REPLACE)
      "####" "#" 25
      Stop
  }
}
